#用来清空全部头部骨骼，以便制作脖子以下模型
headkey="Head"
arm0= bpy.context.selected_objects[0].data
bpy.ops.object.mode_set(mode='EDIT')
bpy.ops.armature.select_all(action='DESELECT')

bpy.ops.object.mode_set(mode='OBJECT')
